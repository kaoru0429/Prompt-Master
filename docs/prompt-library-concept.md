  const compiled = compileTemplate(prompt.content, execution.variables);

  const response = await fetch(`${execution.model.endpoint}/chat/completions`, {
    method: 'POST',

      model: execution.model.name,
      messages: [{ role: 'user', content: compiled }],
      stream: execution.stream
    })
  });

  return execution.stream
    ? handleStreamResponse(response)
    : (await response.json()).choices[0].message.content;
}

### 3.2 è³‡æ–™åº«è¨­è¨ˆ (Database Schema)

```sql
-- Subscriptions Table
CREATE TABLE subscriptions (
    id TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    source TEXT NOT NULL,
    type TEXT CHECK(type IN ('github', 'url', 'local')),
    auto_update BOOLEAN DEFAULT TRUE,
    last_sync TIMESTAMP,
    hash TEXT
);

-- Prompts Table
CREATE TABLE prompts (
    uuid TEXT PRIMARY KEY,
    title TEXT NOT NULL,
    content TEXT NOT NULL,
    source_id TEXT REFERENCES subscriptions(id),
    favorite BOOLEAN DEFAULT FALSE,
    use_count INTEGER DEFAULT 0,
    last_used TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    modified_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Tags Table
CREATE TABLE tags (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT UNIQUE NOT NULL
);

-- Prompt_Tags (Many-to-Many)
CREATE TABLE prompt_tags (
    prompt_uuid TEXT REFERENCES prompts(uuid),
    tag_id INTEGER REFERENCES tags(id),
    PRIMARY KEY (prompt_uuid, tag_id)
);

-- Variables Table
CREATE TABLE variables (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    prompt_uuid TEXT REFERENCES prompts(uuid),
    key TEXT NOT NULL,
    label TEXT,
    type TEXT CHECK(type IN ('text', 'number', 'select', 'multiselect')),
    default_value TEXT,
    options TEXT, -- JSON array for select/multiselect
    required BOOLEAN DEFAULT FALSE
);

-- Execution History
CREATE TABLE executions (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    prompt_uuid TEXT REFERENCES prompts(uuid),
    variables_json TEXT,
    model TEXT,
    response TEXT,
    executed_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
```
